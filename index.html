<h1>-----------app.py----------------------</h1> from flask import Flask, render_template, request, jsonify app = Flask(__name__) # Dictionary containing details about each fruit fruit_details = { 'apple': {'color': 'red', 'taste': 'sweet'}, 'banana': {'color': 'yellow', 'taste': 'sweet'}, 'orange': {'color': 'orange', 'taste': 'sweet and tangy'}, 'grape': {'color': 'green, purple', 'taste': 'sweet'}, 'kiwi': {'color': 'brown', 'taste': 'tangy'}, 'pineapple': {'color': 'yellow', 'taste': 'tangy and sweet'}, 'strawberry': {'color': 'red', 'taste': 'sweet'}, 'watermelon': {'color': 'green', 'taste': 'sweet and juicy'} } @app.route('/') def index(): return render_template('search.html') @app.route('/search',methods=['GET']) def search(): query = request.args.get('query') print(query) # Perform search logic results = [] if query=="" or query is None: return jsonify({'results':[]}) for fruit, details in fruit_details.items(): if query.lower() in fruit.lower(): result = {'name': fruit, 'details': details} results.append(result) print(results) return jsonify({'results': results}) if __name__ == '__main__': app.run(debug=True) <h1>-----search.html(put inside folder name templates!!)---------------</h1> <!DOCTYPE html> <head> <title>Live Search and Dynamic Form Submission</title> <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.5.1/jquery.min.js"></script> </head> <body> <h2>Live Search</h2> <!-- Step 2: Live Search - Create input and search results container --> <input type="text" id="searchInput" placeholder="Search..."> <div id="searchResults"></div> <script> $(document).ready(function(){ // Function to display search results function displayResults(results) { var html = ''; if (results.length > 0) { $.each(results, function(index, result){ html += '<p><strong>' + result.name + '</strong>: ' + result.details.color+' '+ result.details.taste+ '</p>'; }); } else { html = '<p>No results found</p>'; } $('#searchResults').html(html); } // Submit search form $('#searchInput').on('keyup', function(){ var searchText = $(this).val(); if(searchText !== ''){ $.ajax({ url: '/search', method: 'GET', data: {query: searchText}, success: function(response){ displayResults(response.results); } }); } else { $('#searchResults').html(''); } }); }); </script> </body> </html>
